% --- Language and fonts (LuaLaTeX + fontspec + polyglossia) -----------------
% PRD §8.5: Hebrew font = David CLM; English font = Latin Modern Roman.
% FR-20 §16.3: fontspec + polyglossia under LuaLaTeX.
\usepackage{fontspec}
\usepackage{polyglossia}
\setmainlanguage{english}
\setotherlanguage{hebrew}
\setmainfont{Latin Modern Roman}
\newfontfamily\hebrewfont[Script=Hebrew]{David CLM}

% --- Bibliography (biblatex + biber) ----------------------------------------
% docs/PRD_bibliography_and_citations.md §13.1:
%   style=numeric, sorting=none, backend=biber.
\usepackage[style=numeric,sorting=none,backend=biber]{biblatex}
\addbibresource{references.bib}

% --- Index ------------------------------------------------------------------
% intoc is passed to \makeindex (not the package) in TeX Live 2025+,
% which adds the index TOC entry automatically.
\usepackage{imakeidx}
\makeindex[intoc]

% --- Nomenclature -----------------------------------------------------------
% nomencl intoc option adds the nomenclature TOC entry automatically.
% makeindex main.nlo -s nomencl.ist -o main.nls is required between passes.
\usepackage[intoc]{nomencl}
\makenomenclature

% --- Mathematics ------------------------------------------------------------
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{amsthm}

% --- Graphics and figures ---------------------------------------------------
\usepackage{graphicx}
\usepackage{tikz}
\usetikzlibrary{arrows.meta,positioning,shapes.geometric}

% --- Tables -----------------------------------------------------------------
\usepackage{booktabs}
\usepackage{array}

% --- Layout and misc --------------------------------------------------------
\usepackage{xcolor}
\usepackage{xspace}

% --- Headers and footers (fancyhdr) — FR-21 ---------------------------------
% headheight must be ≥13.6pt for 11pt body text (fancyhdr requirement).
\usepackage{fancyhdr}
\setlength{\headheight}{14pt}
\addtolength{\topmargin}{-2pt}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE,RO]{\thepage}
\fancyhead[LO]{\nouppercase{\rightmark}}
\fancyhead[RE]{\nouppercase{\leftmark}}
\fancyfoot[C]{\small Agentic Reasoning for Large Language Models}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.2pt}

% Plain style for chapter-opening pages
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[C]{\small\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%
}

% --- Theorem-like environments — FR-24 / AC §14.2 ---------------------------
\newtheorem{definition}{Definition}[chapter]
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{example}{Example}[chapter]
